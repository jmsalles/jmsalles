# Guia de criação e deploy do Portal Homelab Nginx com Gitea Actions

## 1. Objetivo

Este documento descreve a criação do projeto:

```text
portal-homelab-nginx
```

O objetivo é criar um portal web estático para o homelab JMSalles, com múltiplas páginas, informações dos serviços, checklist, links úteis e topologia do ambiente.

A aplicação será publicada no servidor Docker HML através do Gitea Actions, usando Docker Compose.

URL de acesso:

```text
http://docker-hml.jmsalles.homelab.br:18081
```

IP direto:

```text
http://192.168.31.37:18081
```

---

## 2. Visão geral da arquitetura

```text
Gitea
|
Push na branch main
|
Gitea Actions
|
Runner docker-hml
|
Publica arquivos em /home/jmsalles/docker-apps/portal-homelab-nginx/site
|
docker compose up -d
|
Container Nginx
|
portal-homelab-nginx
|
Porta 18081
```

Fluxo do usuário:

```text
Usuário
|
HTTP
|
docker-hml.jmsalles.homelab.br:18081
|
Container portal-homelab-nginx
|
Nginx
|
Arquivos HTML/CSS/JS
```

| Workflow                       | `.gitea/workflows/deploy.yml`                          |
| URL                            | `http://docker-hml.jmsalles.homelab.br:18081`          |

```text

```

```text
18080 -> app-hello-homelab
18081 -> portal-homelab-nginx
```

Estrutura no Docker HML:

```text
/home/jmsalles/docker-apps/
|
|-- app-hello-homelab
|-- portal-homelab-nginx
```

## 5. Criar repositório no Gitea

Acesse:

```text
https://git.jmsalles.homelab.br
```

Crie o repositório:

| Campo            | Valor                  |
| ---------------- | ---------------------- |
| Owner            | `jmsalles`             |
| Repository name  | `portal-homelab-nginx` |
| Branch principal | `main`                 |
| Visibilidade     | Privado ou público     |

URL esperada:

```text
https://git.jmsalles.homelab.br/jmsalles/portal-homelab-nginx
```

```javascript
function updateClock() {
  const clock = document.getElementById("clock");

  if (!clock) {
    return;
  }

  const now = new Date();

  clock.textContent = "Data e hora local: " + now.toLocaleString("pt-BR");
}

function updateVisitCounter() {
  const counter = document.getElementById("visit-counter");

  if (!counter) {
    return;
  }

  const key = "portalHomelabVisits";
  const current = Number(localStorage.getItem(key) || "0") + 1;

  localStorage.setItem(key, String(current));

  counter.textContent = "Visitas neste navegador: " + current;
}

function loadChecklist() {
  const items = document.querySelectorAll(".check-item");

  items.forEach((item) => {
    const key = "check-" + item.dataset.key;
    item.checked = localStorage.getItem(key) === "true";

    item.addEventListener("change", () => {
      localStorage.setItem(key, String(item.checked));
    });
  });
}

async function loadServices() {
  const container = document.getElementById("services-list");
  const filter = document.getElementById("service-filter");

  if (!container) {
    return;
  }

  try {
    const response = await fetch("/data/servicos.json");
    const services = await response.json();

    function render(list) {
      container.innerHTML = "";

      list.forEach((service) => {
        const card = document.createElement("article");
        card.className = "card";

        card.innerHTML = `
          <h3>${service.name}</h3>
          <p>${service.description}</p>
          <p><strong>Status:</strong> <span class="status-ok">${service.status}</span></p>
          <a href="${service.url}" target="_blank">${service.url}</a>
        `;

        container.appendChild(card);
      });
    }

    render(services);

    if (filter) {
      filter.addEventListener("input", () => {
        const value = filter.value.toLowerCase();

        const filtered = services.filter((service) => {
          return (
            service.name.toLowerCase().includes(value) ||
            service.url.toLowerCase().includes(value) ||
            service.description.toLowerCase().includes(value) ||
            service.status.toLowerCase().includes(value)
          );
        });

        render(filtered);
      });
    }
  } catch (error) {
    container.innerHTML = `<p class="status-error">Erro ao carregar serviços: ${error}</p>`;
  }
}

updateClock();
updateVisitCounter();
loadChecklist();
loadServices();

setInterval(updateClock, 1000);
```

## 19. Criar workflow Gitea Actions

```bash
vi /home/jmsalles/portal-homelab-nginx/.gitea/workflows/deploy.yml
```

Conteúdo:

```yaml
name: Deploy Portal Homelab Nginx

on:
  push:
    branches:
      - main

jobs:
  deploy:
    runs-on: docker-hml

    defaults:
      run:
        shell: sh

    env:
      GIT_SSL_CAINFO: /etc/ssl/certs/ca-certificates.crt
      NODE_EXTRA_CA_CERTS: /etc/ssl/certs/ca-certificates.crt

      DEPLOY_DIR: /deploy/portal-homelab-nginx
      SITE_DIR: /deploy/portal-homelab-nginx/site

      HOST_DEPLOY_DIR: /home/jmsalles/docker-apps/portal-homelab-nginx
      HOST_SITE_DIR: /home/jmsalles/docker-apps/portal-homelab-nginx/site

      CONTAINER_NAME: portal-homelab-nginx
      APP_PORT: 18081

    steps:
      - name: Preparar ambiente do job
        run: |
          apk add --no-cache git nodejs npm bash ca-certificates docker-cli-compose curl rsync
          cp /certs/jmsalles-homelab-rootCA.crt /usr/local/share/ca-certificates/jmsalles-homelab-rootCA.crt
          update-ca-certificates
          git config --global http.sslCAInfo /etc/ssl/certs/ca-certificates.crt
          git config --global http.sslVerify true
          git --version
          node --version
          docker version
          docker compose version

      - name: Checkout do codigo
        uses: actions/checkout@v4

      - name: Validar estrutura do site
        run: |
          echo "Diretorio atual:"
          pwd

          echo "Arquivos do repositorio:"
          find . -maxdepth 4 -type f | sort

          test -f site/index.html || {
            echo "ERRO: site/index.html nao encontrado"
            exit 1
          }

          test -f site/pages/topologia.html || {
            echo "ERRO: pagina de topologia nao encontrada"
            exit 1
          }

          test -f site/assets/img/topologia-homelab.svg || {
            echo "ERRO: imagem SVG da topologia nao encontrada"
            exit 1
          }

          test -f site/assets/css/style.css || {
            echo "ERRO: CSS principal nao encontrado"
            exit 1
          }

          test -f site/assets/js/app.js || {
            echo "ERRO: JavaScript principal nao encontrado"
            exit 1
          }

      - name: Criar estrutura de deploy
        run: |
          mkdir -p "${DEPLOY_DIR}"/{site,logs,config}
          mkdir -p "${SITE_DIR}"/{pages,assets/{css,js,img},data}

          echo "Teste de escrita no /deploy:"
          echo "teste-mount-deploy" > /deploy/teste-mount.txt
          ls -lah /deploy

          echo "Estrutura do deploy:"
          ls -lah "${DEPLOY_DIR}"

      - name: Publicar arquivos estaticos no host
        run: |
          rsync -av --delete site/ "${SITE_DIR}/"

          echo "Arquivos publicados em ${SITE_DIR}:"
          find "${SITE_DIR}" -maxdepth 4 -type f | sort

          echo "Validando index.html no destino:"
          test -f "${SITE_DIR}/index.html" || {
            echo "ERRO: index.html nao foi publicado em ${SITE_DIR}"
            ls -lah "${SITE_DIR}"
            exit 1
          }

      - name: Gerar docker-compose.yml
        run: |
          cat > "${DEPLOY_DIR}/docker-compose.yml" <<EOF
          services:
            portal-homelab-nginx:
              image: nginx:alpine
              container_name: ${CONTAINER_NAME}
              restart: unless-stopped
              ports:
                - "${APP_PORT}:80"
              volumes:
                - ${HOST_SITE_DIR}:/usr/share/nginx/html:ro
          EOF

          cat "${DEPLOY_DIR}/docker-compose.yml"

      - name: Deploy no Docker HML
        run: |
          docker compose \
            --project-name portal-homelab-nginx \
            -f "${DEPLOY_DIR}/docker-compose.yml" \
            pull

          docker compose \
            --project-name portal-homelab-nginx \
            -f "${DEPLOY_DIR}/docker-compose.yml" \
            up -d --force-recreate

      - name: Validar deploy
        run: |
          docker ps --filter "name=${CONTAINER_NAME}"

          docker compose \
            --project-name portal-homelab-nginx \
            -f "${DEPLOY_DIR}/docker-compose.yml" \
            ps

          sleep 3

          echo "Validando arquivos dentro do container:"
          docker exec "${CONTAINER_NAME}" ls -lah /usr/share/nginx/html

          echo "Validando index.html dentro do container:"
          docker exec "${CONTAINER_NAME}" test -f /usr/share/nginx/html/index.html

          echo "Validando pagina inicial:"
          docker exec "${CONTAINER_NAME}" wget -qO- http://127.0.0.1 | grep "Portal Homelab JMSalles"

          echo "Validando pagina de topologia:"
          docker exec "${CONTAINER_NAME}" wget -qO- http://127.0.0.1/pages/topologia.html | grep "Topologia do Homelab JMSalles"
```

## 20. Ajustar Gitea Runner para permitir deploy

No host Docker HML, validar o arquivo do runner:

```bash
vi /home/jmsalles/gitea-runner/config.yaml
```

O bloco `container` deve conter o volume `/home/jmsalles/docker-apps:/deploy:rw`:

```yaml
container:
  network: ""
  privileged: false
  options: "-v /home/jmsalles/gitea-runner/certs:/certs:ro -v /home/jmsalles/docker-apps:/deploy:rw -e GIT_SSL_CAINFO=/certs/jmsalles-homelab-rootCA.crt -e NODE_EXTRA_CA_CERTS=/certs/jmsalles-homelab-rootCA.crt"
  workdir_parent:
  valid_volumes:
    - /home/jmsalles/gitea-runner/certs
    - /home/jmsalles/docker-apps
```

Não adicionar `/var/run/docker.sock` neste campo, porque isso pode gerar erro de mount duplicado.

O socket Docker deve ficar somente no `docker-compose.yml` do runner.

## 28. Troubleshooting

### 28.1 Pipeline falha com erro de permissão no `/deploy`

Validar se o runner tem o volume:

```yaml
-v /home/jmsalles/docker-apps:/deploy:rw
```

Validar no host:

```bash
ls -ld /home/jmsalles/docker-apps
```

Corrigir permissão se necessário:

```bash
sudo chown -R jmsalles:jmsalles /home/jmsalles/docker-apps
```

---

### 28.2 Container retorna 403 Forbidden

Validar se existe `index.html` no destino:

```bash
ls -lah /home/jmsalles/docker-apps/portal-homelab-nginx/site
```

Validar se o `docker-compose.yml` está usando o caminho correto do host:

```bash
cat /home/jmsalles/docker-apps/portal-homelab-nginx/docker-compose.yml
```

O volume correto é:

```yaml
volumes:
  - /home/jmsalles/docker-apps/portal-homelab-nginx/site:/usr/share/nginx/html:ro
```

Não usar `/deploy/...` como host path no compose final, porque `/deploy` existe dentro do container do job, não no host Docker.

---

### 28.3 Pipeline falha ao validar `127.0.0.1:18081`

Dentro do job do Gitea Actions, `127.0.0.1` aponta para o container temporário do job, não para o host Docker.

Por isso a validação correta é:

```bash
docker exec portal-homelab-nginx wget -qO- http://127.0.0.1
```

---

### 28.4 Página não atualiza após push

Validar se o `rsync` atualizou o destino:

```bash
find /home/jmsalles/docker-apps/portal-homelab-nginx/site -maxdepth 4 -type f | sort
```

Recriar container manualmente:

```bash
cd /home/jmsalles/docker-apps/portal-homelab-nginx
```

```bash
docker compose up -d --force-recreate
```
